export async function executeTrade(order) {

  // 1) Check safety filters
  if (!safetyEngine.canTrade()) return rejectTrade();

  // 2) Check concurrent positions
  if (openPositions.count() >= config.maxConcurrent)
      return rejectTrade("MAX_CONCURRENT_REACHED");

  // 3) Check time-of-day
  if (!timeFilter.isWithinWindow()) return rejectTrade("TIME_WINDOW");

  // 4) Live DCA logic

  if (!entry.success) return failTrade(entry);
